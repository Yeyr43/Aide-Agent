# P0 — 裸对话终端 MVP 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 启动一个 Textual 终端窗口，用户输入消息，LLM 流式回复。

**Architecture:** 单进程 Textual App → 异步 worker → LLM Gateway (OpenAI/Ollama) → SSE 流 → RichLog 逐字渲染。LLM Gateway 两个 Provider 共享同一套 SSE 解析逻辑，适配器只负责拼 base_url 和 headers。

**Tech Stack:** Python 3.13+, uv, Textual (RichLog + Input), httpx, Pydantic

## Global Constraints

- Python >= 3.13
- uv 包管理，pyproject.toml
- P0 无测试，P4 补
- 无工具调用、无上下文管理、无记忆演化、无托盘、无命令系统
- Core 接口为纯 async 函数调用，不感知 UI 层
- 所有异步基于 asyncio

```python
    """应用配置。从 config.json 加载。"""

            # 开发时从项目根目录的 config/ 读取
            config_path = Path(__file__).parent / "config.json"

            raise FileNotFoundError(
                f"配置文件不存在: {config_path}\n"
                f"请从 config/config.example.json 复制并修改配置。"
            )

        with open(config_path, "r", encoding="utf-8") as f:

        self.llm = LLMConfig(**data["llm"])
```

- [ ] **Step 1: 创建 core/llm_gateway/provider.py**

```python
"""LLM Gateway — AbstractProvider Protocol + 共享 SSE 解析。

两个 Provider (OpenAI/Ollama) 共享同一套 SSE → token 解析逻辑。
适配器只负责拼 base_url 和 headers。
"""

import json
from typing import AsyncIterator, Protocol

import httpx


class AbstractProvider(Protocol):
    """LLM Provider 协议。

    所有 Provider 必须实现此接口。
    P0 仅需 chat()，P1 扩展 function calling 参数。
    """

    async def chat(self, messages: list[dict[str, str]]) -> AsyncIterator[str]:
        """发送对话消息，流式返回 LLM 生成的 token。

        Args:
            messages: 对话历史，格式 [{"role": "user", "content": "..."}, ...]

        Yields:
            str: 每个 token 字符串（逐 chunk 输出）
        """
        ...


async def _parse_sse_stream(response: httpx.Response) -> AsyncIterator[str]:
    """解析 OpenAI 兼容的 SSE 流，逐 chunk yield token。

    适用 OpenAI Chat Completions API 和 Ollama 兼容端点。
    SSE 格式: data: {"choices":[{"delta":{"content":"token"}}]}\n\n

    Args:
        response: httpx 流式响应对象

    Yields:
        str: 每个 token 字符串
    """
    async for line in response.aiter_lines():
        if not line.startswith("data: "):
            continue

        data_str = line[6:]  # 去掉 "data: " 前缀

        if data_str == "[DONE]":
            break

        try:
            data = json.loads(data_str)
            delta = data["choices"][0].get("delta", {})
            content = delta.get("content", "")
            if content:
                yield content
        except (json.JSONDecodeError, KeyError, IndexError):
            # 跳过格式异常的行（部分模型可能在 stream 中返回非标准字段）
            continue
```

```python
"""OpenAI Provider — 适配 OpenAI Chat Completions API (SSE 流)。"""

from typing import AsyncIterator

import httpx

from .provider import _parse_sse_stream


class OpenAIProvider:
    """OpenAI Chat Completions API 适配器。

    使用 /v1/chat/completions 端点，stream=true。
    """

    def __init__(self, model: str, base_url: str, api_key: str) -> None:
        self.model = model
        self.endpoint = base_url.rstrip("/") + "/chat/completions"
        self.api_key = api_key

    async def chat(self, messages: list[dict[str, str]]) -> AsyncIterator[str]:
        """发送消息并流式返回响应 token。"""
        async with httpx.AsyncClient(timeout=httpx.Timeout(60.0)) as client:
            async with client.stream(
                "POST",
                self.endpoint,
                headers={
                    "Authorization": f"Bearer {self.api_key}",
                    "Content-Type": "application/json",
                },
                json={
                    "model": self.model,
                    "messages": messages,
                    "stream": True,
                },
            ) as response:
                response.raise_for_status()
                async for token in _parse_sse_stream(response):
                    yield token
```

```python
"""LLM Gateway — 统一 LLM 调用入口。

P0 仅提供 create_provider() 工厂函数。
P1 扩展 function calling 支持。
"""

from config.settings import LLMConfig

from .provider import AbstractProvider
from .openai_provider import OpenAIProvider
from .ollama_provider import OllamaProvider

__all__ = ["create_provider", "AbstractProvider"]


def create_provider(config: LLMConfig):
    """根据配置创建对应的 LLM Provider 实例。

    Args:
        config: LLMConfig，包含 provider/model/base_url/api_key

    Returns:
        OpenAIProvider 或 OllamaProvider 实例

    Raises:
        ValueError: 不支持的 provider 类型
    """
    if config.provider == "openai":
        return OpenAIProvider(
            model=config.model,
            base_url=config.base_url,
            api_key=config.api_key,
        )
    elif config.provider == "ollama":
        return OllamaProvider(
            model=config.model,
            base_url=config.base_url,
            api_key=config.api_key,
        )
    else:
        raise ValueError(
            f"不支持的 LLM provider: {config.provider}\n"
            f"P0 支持: openai, ollama"
        )
```

```python
"""用户输入组件 — Enter 发送，空消息不发送。"""

from textual.widgets import Input
from textual.message import Message


class InputBox(Input):
    """用户消息输入框。

    Enter 提交消息，空消息被忽略。
    提交后自动清空输入框。
    """

    class Submitted(Message):
        """用户提交了一条消息。"""

        def __init__(self, text: str) -> None:
            self.text = text
            super().__init__()

    async def _on_submit(self, event: Input.Submitted) -> None:
        """按下 Enter 时触发。空消息不发送。"""
        text = event.value.strip()
        if text:
            self.post_message(self.Submitted(text))
            # Input.Submitted 默认不清空，手动清空
            self.value = ""
```

```python
"""消息流组件 — 基于 RichLog，支持 Rich 标记渲染。"""

from textual.widgets import RichLog


class MessageList(RichLog):
    """聊天消息列表。

    基于 RichLog，自动滚动到底部，支持 Rich markup。
    每条消息占一行或多行。
    """

    def __init__(self) -> None:
        super().__init__(highlight=True, markup=True, wrap=True)
        self._ai_buffer = ""  # 当前 AI 消息的累积文本

    def add_user_message(self, text: str) -> None:
        """添加用户消息（右对齐风格）。"""
        self.write(f"[bold cyan]You:[/bold cyan] {text}")

    def add_ai_chunk(self, chunk: str) -> None:
        """追加 AI 回复的一个 token chunk。

        首次调用时写前缀，后续 chunk 直接追加。
        用 RichLog.write() 的增量特性逐字渲染。
        """
        if not self._ai_buffer:
            # 新 AI 消息开头
            self.write("[bold green]AI:[/bold green] ", nl=False)
        self._ai_buffer += chunk
        self.write(chunk, nl=False)

    def finish_ai_message(self) -> str:
        """结束当前 AI 消息，换行，返回完整响应文本。"""
        self.write("")  # 换行
        full = self._ai_buffer
        self._ai_buffer = ""
        return full

    def add_error(self, text: str) -> None:
        """显示错误消息（红色）。"""
        self.write(f"[bold red]Error:[/bold red] {text}")
```

```python
"""Aide P0 — Textual 聊天终端主应用。

数据流:
  InputBox.Submitted → App.start_chat() → chat_worker()
  → provider.chat(messages) → AsyncIterator[chunk]
  → MessageList.add_ai_chunk(chunk) → 逐字渲染
"""

from textual.app import App, ComposeResult
from textual.worker import Worker, WorkerState
from textual.work import work

from config.settings import Settings
from core.llm_gateway import create_provider
from .widgets.input_box import InputBox
from .widgets.message_list import MessageList


class AideApp(App):
    """Aide Agent P0 — 裸对话终端。"""

    TITLE = "Aide Agent"
    SUB_TITLE = "P0 — 裸对话终端"

    CSS = """
    MessageList {
        height: 1fr;
    }
    InputBox {
        dock: bottom;
        margin: 1;
        height: auto;
    }
    """

    BINDINGS = [
        ("escape", "quit", "退出"),
    ]

    def compose(self) -> ComposeResult:
        """构建 UI 布局：消息区 + 输入框。"""
        yield MessageList(id="messages")
        yield InputBox(
            placeholder="输入消息... (Esc 退出)",
            id="input",
        )

    def on_mount(self) -> None:
        """启动时加载配置并初始化 LLM Provider。"""
        # 加载配置
        try:
            settings = Settings()
            self.provider = create_provider(settings.llm)
        except FileNotFoundError as e:
            self.query_one("#messages", MessageList).add_error(str(e))
            # 配置文件缺失，provider 置 None，后续发消息时拦截
            self.provider = None
        except Exception as e:
            self.query_one("#messages", MessageList).add_error(f"配置加载失败: {e}")
            self.provider = None

        # 对话历史（内存中，P0 不持久化）
        self.conversation: list[dict[str, str]] = []

        # 聚焦输入框
        self.query_one("#input", InputBox).focus()

    def on_input_box_submitted(self, event: InputBox.Submitted) -> None:
        """用户发送消息 → 加入历史 → 启动 worker。"""
        user_text = event.text

        # 显示用户消息
        msg_list = self.query_one("#messages", MessageList)
        msg_list.add_user_message(user_text)

        # 加入对话历史
        self.conversation.append({"role": "user", "content": user_text})

        # 禁用输入，启动 worker
        input_box = self.query_one("#input", InputBox)
        input_box.disabled = True

        if self.provider is None:
            msg_list.add_error("LLM Provider 未初始化，请检查 config/config.json 配置")
            input_box.disabled = False
            input_box.focus()
            return

        self.chat_worker(self.conversation.copy())

    @work(exclusive=True)
    async def chat_worker(self, messages: list[dict[str, str]]) -> None:
        """异步 worker：调用 LLM，流式渲染回复。

        exclusive=True 确保同一时间只有一轮对话在执行。
        """
        msg_list = self.query_one("#messages", MessageList)

        try:
            async for chunk in self.provider.chat(messages):
                msg_list.add_ai_chunk(chunk)

            full_response = msg_list.finish_ai_message()
            self.conversation.append({"role": "assistant", "content": full_response})

        except Exception as e:
            # 流过程中出错：已渲染的部分保留，追加错误提示
            if msg_list._ai_buffer:
                msg_list.finish_ai_message()
            msg_list.add_error(f"LLM 调用失败: {e}")

        finally:
            input_box = self.query_one("#input", InputBox)
            input_box.disabled = False
            input_box.focus()
```
